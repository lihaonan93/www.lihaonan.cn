```python
#!/usr/bin/python
# coding=utf-8
# 处理nginx json格式日志
log = '/var/log/nginx/www.access.log'
i = 0
for lines in open(log, 'r', encoding='utf-8'):
    lines.rstrip('\n')  # 去除行尾 \n 换行符
    src = eval(lines)  # 把字符串转换成字典
    sj = src.get('data')
    ip = src.get('ip')
    rt = src.get('request_time')
    ut = src.get('up_resp_time')
    code = src.get('status')
    rtk = int(rt * 1000)
    if rtk > 800:
        i += 1
        print('日期：{}, ip：{}, 响应时间：{}  {}, 状态码：{}'.format(sj, ip, rt, ut, code))
print('响应时间超过800毫秒的请求有{}个'.format(i))

```
