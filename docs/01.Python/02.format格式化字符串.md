```python
#!/usr/bin/python
# coding=utf-8

name = 'Alex'
age = 20

print(f'我叫{name}, 今年{age}岁了')

print('我叫{0}, 今年{1}岁了'.format(name, age))

print('我叫%s, 今年%d岁了' % (name, age))
```
