```bash
#!/bin/bash
docker run -d --name openresty -p 80:80 -p 443:443 \
    -v /data/nginx/etc/nginx.conf:/etc/openresty/nginx.conf \
    -v /data/nginx/etc/conf.d:/etc/openresty/conf.d \
    -v /data/nginx/ssl:/ssl \
    -v /data/www/:/www \
    -v /data/nginx/logs:/var/log/nginx \
    -e TZ=Asia/Shanghai \
    openresty:1.21.4.1

```
