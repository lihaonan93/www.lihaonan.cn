```bash
[global]
error_log = /var/log/php/php-fpm-error.log
daemonize = no
[www]
user = www-data
group = www-data
listen = 9000

pm = dynamic
pm.max_children = 50
pm.start_servers = 4
pm.min_spare_servers = 1
pm.max_spare_servers = 4
pm.max_requests = 1500

slowlog = /var/log/php/php-slow.log
request_slowlog_timeout = 5
```
